```ts
class EventEmitter {
  public emit(e: EventTypes): string {
    return e.type
  }
}

class Logger {
  public logEvent(eventType: string): void {
    if (eventType.search(/\s/) > -1) throw new Error('hey! no spaces!')
    // ...
  }
}
```
